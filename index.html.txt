    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
        import { getFirestore, collection, addDoc, onSnapshot, deleteDoc, doc } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";
        import { getStorage, ref, uploadBytes, getDownloadURL } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-storage.js";

        const firebaseConfig = {
            apiKey: "AIzaSyDXOljj0H5909ivD-OhMbIVsF_pjzSDtBg",
            authDomain: "pharmacie-samy.firebaseapp.com",
            projectId: "pharmacie-samy",
            storageBucket: "pharmacie-samy.firebasestorage.app",
            messagingSenderId: "178950700829",
            appId: "1:178950700829:web:1d5f72f25352c3fe642bf4"
        };

        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const storage = getStorage(app);

        // نظام الحماية
        window.openAdmin = () => {
            const code = prompt("يرجى إدخال رمز الإدارة:");
            if (code === "pharmaciesami001") {
                document.getElementById('adminPanel').classList.remove('hidden');
            } else {
                alert("الرمز خاطئ!");
            }
        };

        // حفظ المنتج
        window.saveProduct = async () => {
            const name = document.getElementById('pName').value;
            const price = document.getElementById('pPrice').value;
            const file = document.getElementById('pImage').files[0];
            const btn = document.getElementById('btnSave');

            if (!name || !price || !file) return alert("أكمل البيانات أولاً");

            btn.disabled = true; btn.innerHTML = "جاري الرفع...";

            try {
                const imgRef = ref(storage, `products/${Date.now()}_${file.name}`);
                await uploadBytes(imgRef, file);
                const url = await getDownloadURL(imgRef);
                await addDoc(collection(db, "meds"), { name, price, img: url });
                alert("تم بنجاح!");
                location.reload();
            } catch (e) { alert("خطأ! تأكد من تفعيل Storage وتعديل التاريخ"); }
        };

        // عرض المنتجات
        onSnapshot(collection(db, "meds"), (snap) => {
            const grid = document.getElementById('productsGrid');
            grid.innerHTML = '';
            snap.forEach((d) => {
                const p = d.data();
                grid.innerHTML += `
                    <div class="glass p-3 rounded-3xl card-hover shadow-sm border-white">
                        <img src="${p.img}" class="w-full h-40 object-cover rounded-2xl mb-3">
                        <h3 class="font-bold text-gray-800">${p.name}</h3>
                        <p class="text-pink-600 font-bold">${p.price} د.ج</p>
                        <button onclick="deleteProd('${d.id}')" class="text-xs text-red-300 hover:text-red-500 mt-2">حذف</button>
                    </div>
                `;
            });
        });

        window.deleteProd = async (id) => {
            const c = prompt("أدخل الرمز للحذف:");
            if(c === "pharmaciesami001") await deleteDoc(doc(db, "meds", id));
        };
    </script>
